document.addEventListener('DOMContentLoaded', function () { // URL'yi alın const currentURL = window.location.href; // URL'yi "/" karakterine göre ayırın const parts = currentURL.split('/'); // Son parçayı alın const lastPart = parts[parts.length - 1]; // Belirli accordion'ları açmak için son parçaya göre kontrol edin switch (true) { case lastPart.includes('hakkimizda'): case lastPart.includes('yonetim-kurulu'): case lastPart.includes('ust-yonetim'): case lastPart.includes('mali-duyurular'): case lastPart.includes('faaliyet-raporleri'): case lastPart.includes('stratejik-planlar'): case lastPart.includes('performans-programlari'): case lastPart.includes('arabuluculuk-komisyonu'): case lastPart.includes('insan-kaynaklari'): case lastPart.includes('teskilat-semasi'): case lastPart.includes('yerleskelerimiz'): highlightSelectedItem("ust-yonetim"); break; default: // İlgili accordion bulunamazsa bir şey yapmayın break; } function highlightSelectedItem(currentPath) { const allItems = document.querySelectorAll('.accordion-flush a'); allItems.forEach(item => { if (item.getAttribute('href').includes(currentPath)) { item.classList.add('selected-item'); } else { item.classList.remove('selected-item'); } }); } });
